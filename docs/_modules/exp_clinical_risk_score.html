"""Clinical risk-score experiment implementation.

This module builds an interpretable logistic-regression risk score over a
bounded set of numeric and low-cardinality categorical features, converts the
linear predictor into a standardized clinical z-score, and benchmarks the
criterion with ROC curves.
"""

    def run(self, data: pd.DataFrame, config: Dict[str, Any]) -> None:
        """Builds a clinical z-score and evaluates ROC performance."""
        target_col: str = str(config.get("score_target_column", "ana_dura"))
        positive_label: str = str(config.get("score_positive_label", "Buscada positivo"))
        negative_label: str = str(config.get("score_negative_label", "Buscada negativo"))
        max_samples: int = int(config.get("score_max_samples", 4000))
        max_feature_cardinality: int = int(config.get("score_max_feature_cardinality", 12))
        cv_splits: int = int(config.get("score_cv_splits", 5))
        benchmark_model: str = str(config.get("score_benchmark_model", "both"))
        top_features: int = int(config.get("score_top_features", 12))
        xgboost_estimators: int = int(config.get("score_xgboost_estimators", 80))

        modeling_df: pd.DataFrame = self._prepare_modeling_frame(
            data=data,
            target_col=target_col,
            positive_label=positive_label,
            negative_label=negative_label,
            max_samples=max_samples,
            max_feature_cardinality=max_feature_cardinality,
        )

        y: np.ndarray = np.where(modeling_df[target_col] == positive_label, 1, 0)
        X: pd.DataFrame = modeling_df.drop(columns=[target_col])
        cat_cols: List[str] = X.select_dtypes(include=["object", "category", "string"]).columns.tolist()
        num_cols: List[str] = X.select_dtypes(include=[np.number]).columns.tolist()

        if not cat_cols and not num_cols:
            raise ValueError("The risk-score experiment found no eligible predictive features after preprocessing.")

        preprocessor = self._build_preprocessor(cat_cols=cat_cols, num_cols=num_cols)
        effective_splits: int = self._resolve_cv_splits(y=y, requested_splits=cv_splits)
        cv = StratifiedKFold(n_splits=effective_splits, shuffle=True, random_state=42)

        logistic_pipeline = Pipeline(
            steps=[
                ("preprocessor", preprocessor),
                (
                    "model",
                    LogisticRegression(
                        max_iter=1000,
                        solver="liblinear",
                        class_weight="balanced",
                        random_state=42,
                    ),
                ),
            ]
        )

        logistic_eval: ModelEvaluation = self._evaluate_logistic_pipeline(
            pipeline=logistic_pipeline,
            X=X,
            y=y,
            cv=cv,
        )
        evaluations: List[ModelEvaluation] = [logistic_eval]

        if benchmark_model in {"xgboost", "both"}:
            xgb_pipeline = Pipeline(
                steps=[
                    ("preprocessor", self._build_preprocessor(cat_cols=cat_cols, num_cols=num_cols)),
                    (
                        "model",
                        XGBClassifier(
                            n_estimators=xgboost_estimators,
                            max_depth=3,
                            learning_rate=0.08,
                            subsample=0.8,
                            colsample_bytree=0.8,
                            eval_metric="logloss",
                            random_state=42,
                            n_jobs=1,
                        ),
                    ),
                ]
            )
            evaluations.append(
                self._evaluate_probability_pipeline(
                    pipeline=xgb_pipeline,
                    X=X,
                    y=y,
                    cv=cv,
                    model_name="XGBoost Benchmark",
                    score_label="Predicted positive probability",
                )
            )

        fitted_logistic = clone(logistic_pipeline)
        fitted_logistic.fit(X, y)
        coefficient_table: pd.DataFrame = self._build_coefficient_table(
            pipeline=fitted_logistic,
            top_features=top_features,
        )
        metrics_table: pd.DataFrame = self._build_metrics_table(evaluations=evaluations)
        zscore_summary: pd.DataFrame = self._build_zscore_summary(
            z_scores=logistic_eval.scores,
            y_true=logistic_eval.y_true,
            positive_label=positive_label,
            negative_label=negative_label,
        )

        self.latex_table = "\n\n".join(
            [
                metrics_table.to_latex(
                    index=False,
                    caption="Cross-Validated ROC Performance for the Clinical Criterion",
                    label="tab:clinical_roc",
                ),
                zscore_summary.to_latex(
                    index=False,
                    caption="Clinical Z-Score Distribution by Diagnostic Outcome",
                    label="tab:clinical_zscore_distribution",
                ),
                coefficient_table.to_latex(
                    index=False,
                    caption="Most Influential Standardized Components of the Clinical Z-Score",
                    label="tab:clinical_zscore_coefficients",
                ),
            ]
        )
        self.plotly_figure = self._build_roc_figure(
            evaluations=evaluations,
            logistic_eval=logistic_eval,
            positive_label=positive_label,
            negative_label=negative_label,
        )



    def _prepare_modeling_frame(
        self,
        data: pd.DataFrame,
        target_col: str,
        positive_label: str,
        negative_label: str,
        max_samples: int,
        max_feature_cardinality: int,
    ) -> pd.DataFrame:
        """Filters the dataset to the binary criterion and retains interpretable features."""
        if target_col not in data.columns:
            raise ValueError(f"The risk-score experiment requires the target column '{target_col}'.")

        df: pd.DataFrame = data.copy()
        df[target_col] = df[target_col].astype("string")
        df = df[df[target_col].isin([positive_label, negative_label])].reset_index(drop=True)

        if df.empty:
            raise ValueError(
                "No rows matched the requested positive/negative labels for the clinical risk-score experiment."
            )

        if len(df) > max_samples:
            df = (
                df.groupby(target_col, group_keys=False)
                .apply(
                    lambda frame: frame.sample(
                        n=max(1, int(round(max_samples * len(frame) / len(df)))),
                        random_state=42,
                    )
                )
                .reset_index(drop=True)
            )
            if len(df) > max_samples:
                df = df.sample(n=max_samples, random_state=42).reset_index(drop=True)

        feature_cols: List[str] = []
        for column in df.columns:
            if column in {target_col, "id_pacie"}:
                continue

            if pd.api.types.is_numeric_dtype(df[column]):
                feature_cols.append(column)
                continue

            unique_count: int = int(df[column].nunique(dropna=False))
            if 2 <= unique_count <= max_feature_cardinality:
                feature_cols.append(column)

        if not feature_cols:
            raise ValueError(
                "No eligible features remained for the clinical risk-score model. "
                "Increase --score-max-feature-cardinality or review the dataset schema."
            )

        modeling_df: pd.DataFrame = df[[target_col] + feature_cols].copy()
        for column in modeling_df.select_dtypes(include=["object", "category", "string"]).columns:
            modeling_df[column] = modeling_df[column].astype("string").fillna("Missing").astype(str)

        return modeling_df

    def _build_preprocessor(self, cat_cols: Sequence[str], num_cols: Sequence[str]) -> ColumnTransformer:
        """Creates the shared preprocessing graph for linear and tree models."""
        transformers: List[tuple[str, Any, Sequence[str]]] = []
        if cat_cols:
            transformers.append(
                (
                    "cat",
                    Pipeline(
                        steps=[
                            ("imputer", SimpleImputer(strategy="constant", fill_value="Missing")),
                            ("encoder", OneHotEncoder(handle_unknown="ignore", sparse_output=True)),
                        ]
                    ),
                    list(cat_cols),
                )
            )
        if num_cols:
            transformers.append(
                (
                    "num",
                    Pipeline(
                        steps=[
                            ("imputer", SimpleImputer(strategy="median")),
                            ("scaler", StandardScaler()),
                        ]
                    ),
                    list(num_cols),
                )
            )
        return ColumnTransformer(transformers=transformers)

    def _resolve_cv_splits(self, y: np.ndarray, requested_splits: int) -> int:
        """Caps the CV fold count by the smallest class size."""
        class_counts = np.bincount(y)
        smallest_class: int = int(class_counts.min()) if len(class_counts) > 1 else 0
        effective_splits: int = min(requested_splits, smallest_class)
        if effective_splits < 2:
            raise ValueError("At least two samples per class are required to compute cross-validated ROC curves.")
        return effective_splits

    def _evaluate_logistic_pipeline(
        self,
        pipeline: Pipeline,
        X: pd.DataFrame,
        y: np.ndarray,
        cv: StratifiedKFold,
    ) -> ModelEvaluation:
        """Evaluates the logistic model and standardizes its linear predictor as a z-score."""
        linear_scores: np.ndarray = np.zeros(len(X), dtype=float)

        for train_idx, test_idx in cv.split(X, y):
            fold_pipeline = clone(pipeline)
            fold_pipeline.fit(X.iloc[train_idx], y[train_idx])
            linear_scores[test_idx] = fold_pipeline.decision_function(X.iloc[test_idx])

        z_scores: np.ndarray = self._standardize_scores(linear_scores)
        return self._build_evaluation(
            model_name="Clinical Logistic Z-Score",
            score_label="Standardized logistic z-score",
            y_true=y,
            scores=z_scores,
        )

    def _evaluate_probability_pipeline(
        self,
        pipeline: Pipeline,
        X: pd.DataFrame,
        y: np.ndarray,
        cv: StratifiedKFold,
        model_name: str,
        score_label: str,
    ) -> ModelEvaluation:
        """Evaluates a probability-based classifier with out-of-fold scores."""
        probabilities: np.ndarray = np.zeros(len(X), dtype=float)

        for train_idx, test_idx in cv.split(X, y):
            fold_pipeline = clone(pipeline)
            fold_pipeline.fit(X.iloc[train_idx], y[train_idx])
            probabilities[test_idx] = fold_pipeline.predict_proba(X.iloc[test_idx])[:, 1]

        return self._build_evaluation(
            model_name=model_name,
            score_label=score_label,
            y_true=y,
            scores=probabilities,
        )

    def _build_evaluation(
        self,
        model_name: str,
        score_label: str,
        y_true: np.ndarray,
        scores: np.ndarray,
    ) -> ModelEvaluation:
        """Computes ROC metrics and the Youden operating threshold."""
        fpr, tpr, thresholds = roc_curve(y_true, scores)
        auc: float = float(roc_auc_score(y_true, scores))
        best_idx: int = int(np.argmax(tpr - fpr))
        return ModelEvaluation(
            model_name=model_name,
            score_label=score_label,
            y_true=y_true,
            scores=scores,
            auc=auc,
            fpr=fpr,
            tpr=tpr,
            thresholds=thresholds,
            best_threshold=float(thresholds[best_idx]),
            sensitivity=float(tpr[best_idx]),
            specificity=float(1.0 - fpr[best_idx]),
        )

    def _build_metrics_table(self, evaluations: Sequence[ModelEvaluation]) -> pd.DataFrame:
        """Creates a concise ROC operating summary."""
        return pd.DataFrame(
            [
                {
                    "Model": evaluation.model_name,
                    "AUC": evaluation.auc,
                    "Optimal_Threshold": evaluation.best_threshold,
                    "Sensitivity": evaluation.sensitivity,
                    "Specificity": evaluation.specificity,
                }
                for evaluation in evaluations
            ]
        )

    def _build_zscore_summary(
        self,
        z_scores: np.ndarray,
        y_true: np.ndarray,
        positive_label: str,
        negative_label: str,
    ) -> pd.DataFrame:
        """Summarizes the clinical z-score per outcome class."""
        rows: List[Dict[str, Any]] = []
        for class_value, label in [(0, negative_label), (1, positive_label)]:
            class_scores = z_scores[y_true == class_value]
            rows.append(
                {
                    "Outcome": label,
                    "N": int(len(class_scores)),
                    "Mean_Z": float(np.mean(class_scores)),
                    "Median_Z": float(np.median(class_scores)),
                    "Std_Z": float(np.std(class_scores)),
                    "P25_Z": float(np.quantile(class_scores, 0.25)),
                    "P75_Z": float(np.quantile(class_scores, 0.75)),
                }
            )
        return pd.DataFrame(rows)

    def _build_coefficient_table(self, pipeline: Pipeline, top_features: int) -> pd.DataFrame:
        """Extracts the strongest logistic components behind the clinical z-score."""
        preprocessor: ColumnTransformer = pipeline.named_steps["preprocessor"]
        model: LogisticRegression = pipeline.named_steps["model"]
        feature_names: np.ndarray = preprocessor.get_feature_names_out()
        coefficients: np.ndarray = model.coef_.ravel()
        coef_df: pd.DataFrame = pd.DataFrame(
            {
                "Feature": feature_names,
                "Coefficient": coefficients,
                "Odds_Ratio": np.exp(coefficients),
                "Absolute_Coefficient": np.abs(coefficients),
            }
        )
        coef_df = coef_df.sort_values(by="Absolute_Coefficient", ascending=False).head(top_features)
        return coef_df.drop(columns=["Absolute_Coefficient"]).reset_index(drop=True)

    def _build_roc_figure(
        self,
        evaluations: Sequence[ModelEvaluation],
        logistic_eval: ModelEvaluation,
        positive_label: str,
        negative_label: str,
    ) -> go.Figure:
        """Builds a combined ROC and z-score distribution figure."""
        figure = make_subplots(
            rows=1,
            cols=2,
            subplot_titles=("ROC Curves", "Clinical Z-Score Distribution"),
        )

        for evaluation in evaluations:
            figure.add_trace(
                go.Scatter(
                    x=evaluation.fpr,
                    y=evaluation.tpr,
                    mode="lines",
                    name=f"{evaluation.model_name} (AUC={evaluation.auc:.3f})",
                    hovertemplate=(
                        "Model: " + evaluation.model_name + "<br>"
                        "False Positive Rate: %{x:.3f}<br>"
                        "True Positive Rate: %{y:.3f}<extra></extra>"
                    ),
                ),
                row=1,
                col=1,
            )

        figure.add_trace(
            go.Scatter(
                x=[0.0, 1.0],
                y=[0.0, 1.0],
                mode="lines",
                name="Random baseline",
                line=dict(dash="dash", color="gray"),
                hoverinfo="skip",
            ),
            row=1,
            col=1,
        )

        negative_scores = logistic_eval.scores[logistic_eval.y_true == 0]
        positive_scores = logistic_eval.scores[logistic_eval.y_true == 1]
        figure.add_trace(
            go.Histogram(
                x=negative_scores,
                nbinsx=30,
                name=f"{negative_label} z-score",
                opacity=0.65,
                histnorm="probability density",
            ),
            row=1,
            col=2,
        )
        figure.add_trace(
            go.Histogram(
                x=positive_scores,
                nbinsx=30,
                name=f"{positive_label} z-score",
                opacity=0.65,
                histnorm="probability density",
            ),
            row=1,
            col=2,
        )
        figure.add_vline(
            x=logistic_eval.best_threshold,
            line_dash="dot",
            line_color="black",
            annotation_text=f"Optimal z={logistic_eval.best_threshold:.2f}",
            row=1,
            col=2,
        )

        figure.update_xaxes(title_text="False Positive Rate", row=1, col=1)
        figure.update_yaxes(title_text="True Positive Rate", row=1, col=1)
        figure.update_xaxes(title_text="Clinical z-score", row=1, col=2)
        figure.update_yaxes(title_text="Density", row=1, col=2)
        figure.update_layout(
            title="Clinical Risk Z-Score ROC Benchmark",
            barmode="overlay",
            legend_title="Criterion",
        )
        return figure

    @staticmethod
    def _standardize_scores(scores: Iterable[float]) -> np.ndarray:
        """Standardizes continuous scores into z-units."""
        score_array: np.ndarray = np.asarray(list(scores), dtype=float)
        std: float = float(np.std(score_array))
        if std == 0.0:
            return np.zeros_like(score_array)
        return (score_array - float(np.mean(score_array))) / std
